#include <stdio.h>
#include <stdlib.h>

int main(){
    int *arr,i;

    arr = (int *) malloc(5*sizeof(int));

    printf("Digite os elementos do array");
    for(i=0;i<5;i++){
        printf("%d- ",i+1);
        scanf(" %d",arr+i);
    }

    for(i=0;i<5;i++){
        printf("%d, ",*(arr+i));
    }
}

#include <stdio.h>
#include <stdlib.h>

int main(){
    int *arr,n,i;
    printf("Digite o valor do comprimento do vetor: ");
    scanf("%d",&n);

    arr = (int *) malloc(n*sizeof(int));

    for(i=0;i<n;i++){
        printf("%d- ",i+1);
        scanf("%d",arr+i);
    }

    for(i=0;i<n;i++){
        printf("%d, ",arr[i]);
    }
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(){
    char *str;
    int i,n;

    printf("Digite o tamanho da string: ");
    scanf("%d",&n);

    str = (char *) malloc(n*sizeof(char));
    setbuf(stdin,NULL);
    fgets(str,n,stdin);

    str[strcspn(str,"\r\n")] = 0;

    for(i=0;i<n;i++){
        if(!(str[i]=='a'||str[i]=='e'||str[i]=='i'||str[i]=='o'||str[i]=='u')){
            printf("%c",str[i]);
        }
    }
    free(str);
    return 0;
