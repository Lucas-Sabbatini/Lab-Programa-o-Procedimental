#include <stdio.h>
#include <stdlib.h>

int main(){
    int *arr,i;

    arr = (int *) malloc(5*sizeof(int));

    printf("Digite os elementos do array");
    for(i=0;i<5;i++){
        printf("%d- ",i+1);
        scanf(" %d",arr+i);

    for(i=0;i<5;i++){

#include <stdio.h>
#include <stdlib.h>

int main(){
    int *arr,n,i;
    printf("Digite o valor do comprimento do vetor: ");
    scanf("%d",&n);

    arr = (int *) malloc(n*sizeof(int));

    for(i=0;i<n;i++){
        printf("%d- ",i+1);
        scanf("%d",arr+i);
    }

    for(i=0;i<n;i++){
        printf("%d, ",arr[i]);
    }
}
