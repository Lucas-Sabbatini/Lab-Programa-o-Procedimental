#include <stdio.h>
#include <math.h>
#define PI 3.1415

int vol(int raio);
int main()
{
    int param;
    scanf("%d", &param);
    printf("O volume da esfera eh: %d",vol(param));
    return 0;
}

int vol(int raio){
   return (4/3)*PI*raio*raio*raio;
}
