#include <stdio.h>

int soma (int a, int b);

int main() {
    int p1,p2;
    scanf("%d %d", &p1, &p2);
    printf("A soma dos numeros entre %d e %d eh %d", p1, p2, soma(p1, p2));
    return 0;
}

int soma (int a, int b) {
    int s = 0;
    if (a>b){
        b ++;
        for (b; b < a; b++){
            s += b;
        }
    }
    else if (b>a){
        a ++;
        for (a; a < b; a++){
            s += a;
        }
    }
    return s;
}

{
    int n,nPrimos =0;
    scanf("%d",&n);
    for(int j=2; j<=n; j++){
        nPrimos += primo(j);
    }
    printf("O numeros de primos ate n eh: %d\n",nPrimos);
    
    return 0;
}

int primo(int a){
    int stop = a/2;
    if((a==3)||(a==2)||(a==5)||(a==7)||(a==11)){
        return 1;
    }
    if((a==0)||(a==1)||!(a%3)||!(a%5)||!(a%7)||!(a%11)){
        return 0;
    }
    for(int i = 3;i<=stop;i+=2) {
        if(!(a%i)){
            return 0;
        }
    }
    return 1;
}

int piramide(int a){

    int sinit = 1;
    for(int i=0;i<a;i++) {
        for(int e=0;e<einit;e++){
            printf(" ");
        }

        }
        einit--;
        sinit+=2;
