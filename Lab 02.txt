Exercício 1----------

#include <stdio.h>


int main()
{
    float x,y;
    printf("Digite dois numeros: ");
    scanf("%f %f",&x,&y);
    if (x>y){
        printf("O maior numero eh: %f",x);
    }
    else if (x==y){
        printf("Os dois numeros sao iguais");
    }
    else{
        printf("O maior numero eh: %f",y);
    }
    return 0;
}
