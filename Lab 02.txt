    if(a==0){
        printf("Nao eh equacao do segundo grau");
    }
    else{
        delta = (b*b) - (4*a*c);
        if(delta<0){
            printf("Nao existem raizes reais");
        }
        else if(delta == 0){
            x1 = (-1*b)/2*a;
            printf("A equacao so tem uma raiz real que eh: %f",x1);
        }
        else{
            x1 = (-1*b+sqrt(delta))/2*a;
            x2 = (-1*b-sqrt(delta))/2*a;
            printf("As duas raizes sao: %f %f",x1,x2);
        }

    }

    return 0;
}
