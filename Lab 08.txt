    int a,b;
    printf("Digite os valores de a e b");
    scanf("%d %d",&a,&b);
    troca(&a,&b);
    
    printf("Valor de a: %d Valor de b: %d",a,b);

}

void troca(int *a,int *b){
    int temp = *a;
    *a = *b;
    *b = temp;
}

int ordenado(int *a,int *b,int *c){
    int maior,meio,menor;
    if(*a>*b){
        if(*c>*a){
            // c maior a segundo maior b menor
            maior = *c;
            meio = *a;
            menor = *b;
            
        }
        else if(*c>*b){
            //a maior c segundo maior e b menor
            maior = *a;
            meio = *c;
            menor = *b;
        }
        else{
            //a maior b segundo maior e c menor
            maior = *a;
            meio = *b;
            menor = *c;
        }
        *a = menor;
        *b= meio;
        *c= maior;
        return 0;
    }
    else if(*b>*a){
        if(*c>*b){
            // c maior b segundo maior a menor
            maior = *c;
            meio = *b;
            menor = *a;
        }
        else if(*c>*a){
            //b maior c segundo maior e a menor
            maior = *b;
            meio = *c;
            menor = *a;
        }
        else{
            //b maior a segundo maior e c menor
            maior = *b;
            meio = *a;
            menor = *c;
        }
        *a = menor;
        *b= meio;

        return 1;
