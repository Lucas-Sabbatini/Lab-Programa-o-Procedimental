#include <stdio.h>

void troca(int *a,int *b);
int main(){
    int a,b;
    printf("Digite os valores de a e b");
    scanf("%d %d",&a,&b);
    troca(&a,&b);
    
    printf("Valor de a: %d Valor de b: %d",a,b);
    return 0;
}

void troca(int *a,int *b){
    int temp = *a;
    *a = *b;
    *b = temp;
}

void maior(int *a,int *b);

int main(){
    int a,b;
    printf("Digite os valores de a e b: ");
    scanf("%d %d",&a,&b);
    maior(&a,&b);
    printf("O valor de a eh: %d e o valor de b eh: %d",a,b);
    return 0;
}


void maior(int *a,int *b){
    if(*a<*b){
        int troca;
        troca = *a;
        *a = *b;
        *b = troca;
    }
}

#include <stdio.h>

int soma(int *a,int *b);

int main(){
    int a,b,c;
    printf("Digite os dois valores de a e b respectivamente: ");
    scanf("%d %d",&a,&b);
    c = soma(&a,&b);
    printf("O dobro de a eh %d, o dobro de b eh %d, e a soma dos dobros eh %d",a,b,c);
    return 0;
}

int soma(int *a,int *b){
    *a *=2;
    *b *=2;
    return *a + *b;
}
