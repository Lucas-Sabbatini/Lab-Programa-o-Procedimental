        scanf("%d", &vetor[i]);
    }
    for(int j=0;j<10;j++){
        
        if(primo(vetor[j])==1){
            printf("%d- %d eh primo\n",j,vetor[j]);
        }
    }
        
  
    
    return 0;
}

int primo(int a){
    int stop = a/2;
    if((a==3)||(a==2)||(a==5)||(a==7)||(a==11)){
        return 1;
    }
    if((a==0)||(a==1)||!(a%2)||!(a%3)||!(a%5)||!(a%7)||!(a%11)){
        return 0;
    }
    for(int i = 3;i<=stop;i+=2) {
        if(!(a%i)){
            return 0;
        }
    }
    return 1;
}

Exercício 8------------------

#include <stdio.h>

int main()
{
    int ent,rep=0,vetor[10];
    for(int i = 0; i < 10; i++){
        do{
            scanf("%d", &ent);
            rep =0;
            for(int j = 0; j < i; j++){
                if(vetor[j] == ent){
                    printf("Numeros repetidos, digite outro numero\n");
                    rep=1;
                }
            }
            if(!rep)
                vetor[i] = ent;
        }while(rep);
    }
    return 0;
}

#include <stdio.h>

int main(){
    int count=0,matriz[4] [4];
    printf("Digite os elementos da matriz");
    for(int i=0;i<4;i++){
        for(int j=0;j<4;j++){
            scanf("%d",&matriz[i][j]);
        }
    }

    for(int i=0;i<4;i++){
        for(int j=0;j<4;j++){
            if (matriz[i][j]>10){
                count++;
                printf("%d  [%d][%d] eh maior que 10",matriz[i][j],i,j);
            }
        }
    }

    printf("%d numeros sao maiores que 10",count);

}
