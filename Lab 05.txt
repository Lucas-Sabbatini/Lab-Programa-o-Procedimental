#include <stdio.h>

int main(){
    int mai,mei,maior,menor,vetor[5];
    printf("Digite 5 valores inteiros");
    for(int i=0;i<5;i++){
        scanf("%d",&vetor[i]);
    }
    maior = vetor[4];
    menor = vetor[4];
    for(int j=0;j<5;j++){
        if(maior<=vetor[j]){
            maior = vetor[j];
            mai = j;
        }
        else if(menor>=vetor[j]){
            menor = vetor[j];
            mei = j;
        }
    }
    
    printf("O maior numero esta na posicao %d e o menor numero esta na posicao %d",mai,mei);
    
}
