Exercício 2---------------

#include <stdio.h>
#include <stdlib.h>

int main(){
	int i=1;
	for(i;i<=100;i++){
		printf("%d\n",i);
	}
	i=1;
	while(i<=100){
		printf("%d\n",i++);
	}
	i=1;
	do{
		printf("%d\n",i++);
	}while(i<=100);
}
