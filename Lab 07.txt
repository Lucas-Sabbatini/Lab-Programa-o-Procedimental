#include <stdio.h>
#include <string.h>

struct pessoa{
    char nome[20];
    int idade;
    char endereco[50];
};

int main(){

    struct pessoa p1;
    printf("Digite o seu nome");
    gets(p1.nome);
    printf("Agora digite a sua idade: ");
    scanf("%d",&p1.idade);
    printf("Por ultimo o seu endereco: ");
    setbuf(stdin,NULL);
    gets(p1.endereco);


    printf("\n\n\nNome: %s\nIdade: %d\nEndereco: %s",p1.nome,p1.idade,p1.endereco);
}
